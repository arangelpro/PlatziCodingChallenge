define("HOUR_IN_SECONDS", 3600);
define("MINUTE_IN_SECONDS", 60);
function calculateSeconds($hour, $minute) {
    $seconds = $hour * HOUR_IN_SECONDS + $minute * MINUTE_IN_SECONDS;
    print("Los segundos son: ", $seconds)
    return $seconds;
}
